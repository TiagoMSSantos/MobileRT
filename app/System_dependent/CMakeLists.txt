if( DEFINED ANDROID_ABI )
	file( GLOB_RECURSE MOBILE_DEPENDENT_SOURCES "${CMAKE_CURRENT_SOURCE_DIR}/Android_JNI/**.cpp" )
else()
	file( GLOB MOBILE_DEPENDENT_SOURCES "${CMAKE_CURRENT_SOURCE_DIR}/Linux/**.cpp" )
	if( "$ENV{QT_ACCESSIBILITY}" STREQUAL "1" )
		file( GLOB_RECURSE MOBILE_DEPENDENT_SOURCES_QT "${CMAKE_CURRENT_SOURCE_DIR}/Linux/QT/MobileRT/**.cpp" )
	else()
		file( GLOB_RECURSE MOBILE_DEPENDENT_SOURCES_GTK "${CMAKE_CURRENT_SOURCE_DIR}/Linux/GTK/**.cpp" )
	endif()
endif()

set( MOBILE_DEPENDENT_SOURCES
"${MOBILE_DEPENDENT_SOURCES};${MOBILE_DEPENDENT_SOURCES_QT};${MOBILE_DEPENDENT_SOURCES_GTK}"
	 CACHE STRING "CUSTOM" FORCE )
